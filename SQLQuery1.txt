create Table AirProducts(Id int identity(1,1) NOT NULL,Name varchar(30),Price float,IsBooked bit NOT NULL,IsSaved bit NOT NULL,DepartureTime varchar(15),ArrivalTime varchar(15),PRIMARY KEY(Id))
create Table ActivityProducts(Id int identity(1,1) NOT NULL,Name varchar(30),Price float,IsBooked bit NOT NULL,IsSaved bit NOT NULL,StartTime varchar(10),TotalTime varchar(10),PRIMARY KEY(Id))
create Table HotelProducts(Id int identity(1,1) NOT NULL,Name varchar(30),Price float,IsBooked bit NOT NULL,IsSaved bit NOT NULL,RoomType varchar(10),CheckIn varchar(10),CheckOut varchar(10),Rating varchar(3),PRIMARY KEY(Id))
create Table CarProducts(Id int identity(1,1) NOT NULL,Name varchar(30),Price float,IsBooked bit NOT NULL,IsSaved bit NOT NULL,StartTime varchar(10),EndTime varchar(10)PRIMARY KEY(Id))
